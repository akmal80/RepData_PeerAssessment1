activity$wd[activity$weekday=="Wednesday"] <- "weekday"

activity$wd <- factor(activity$wd)

mean_inter_wd <- aggregate(activity$steps, by=list(Interval=activity$interval, weekday = activity$wd), FUN=mean,  na.rm=TRUE)
